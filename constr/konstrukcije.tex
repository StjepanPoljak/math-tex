\noindent\newline Važna opservacija je takoðer da vrijedi $\omega_{k}=(\omega_1)^k$ (što se može lako pokazati po DeMoivreovoj formuli). \begin{eqnarray*}
\omega_{n-k}&=&\cos{2\pi\frac{n-k}{n}}+i\sin{2\pi\frac{n-k}{n}}\\

\end{eqnarray*}

\noindent\newline Koristeæi èinjenicu kako je $\cos{2\pi-x}=\cos{2\pi}\cos{x}+\sin{2\pi}\sin{x}=\cos{x}$ i $\sin{2\pi-x}=\sin{2\pi}\cos{x}-\sin{x}\cos{2\pi}=-\sin{x}$, imamo:

\begin{equation*}
\omega_{n-k}=\cos{2\pi\frac{k}{n}}-i\sin{2\pi\frac{k}{n}}=\overline{\omega_{k}}.
\end{equation*}

\noindent\newline Iz ovoga slijedi još bitnija èinjenica, a to je da je $\omega_{k}+\omega_{n-k}=2\cos{2\pi\frac{k}{n}}$. \begin{equation*}
(x-1)(x^{n-1}+x^{n-2}+\ldots+x+1)=0.
\end{equation*}

\noindent\newline Dakle, mora vrijediti $x^{n-1}+x^{n-2}+\ldots+x+1=0$ i kako smo argumentirali da je $\omega_{k}=(\omega_1)^{k}$ ($\omega_1$ æemo odsada, upravo zbog ovog svojstva, oznaèavati samo s $\omega$), mora vrijediti i $\omega^{n-1}+\omega^{n-2}+\ldots+\omega+1=0$, tj. vrijedi sljedeæa lema:

\noindent\newline{\bf Lema.} Za temeljne korijene reda $n$ vrijedi:

\begin{equation*}
\sum_{k=0}^{n-1}{\omega^k}=0.
\end{equation*}

\noindent\newline Primijetimo da isto tako, onda vrijedi i:

\begin{equation*}
\sum_{k=1}^{n-1}{\omega^k}=-1.
\end{equation*}

\noindent\newline Vratimo se na naš peterokut. Oznaèimo $x_1=\omega_1+\omega_4$ i $x_2=\omega_2+\omega_3$. Po prethodnoj lemi oèito je kako vrijedi $x_1+x_2=-1$. Isto tako, uzmemo li umnožak ova dva realna broja, dobivamo:

\begin{equation*}

\end{equation*}

\noindent\newline Malo se igramo s oznakama, jer rekli smo kako je zapravo $\omega_k\omega_l=\omega^k\omega^l=\omega^{k+l}=\omega_{k+l}$ (opet se lako dokaže po formuli za množenje kompleksnih brojeva u trigonometrijskom obliku, ukoliko je potrebno daljnje opravdanje). Stoga imamo:

\begin{equation*}

\end{equation*}

\noindent\newline Oèito je $\omega^6=\omega^{5+1}=\omega^5\omega^1=\omega^0\omega^1=\omega$, a isto tako i $\omega_7=\omega_2$. Stoga je:

\begin{equation*}

\end{equation*}

\noindent\newline Sada znamo umnožak i zbroj od $x_1$ i $x_2$ pa njihove vrijednosti možemo pronaæi pomoæu kvadratne jednadžbe $x^2-(x_1+x_2)x+x_1 x_2=0$. Imamo, dakle, $x^2+x-1=0$. Rješenja su tada:

\begin{equation*}

\end{equation*}

\begin{equation*}

\end{equation*}

\noindent\newline No, rekli smo kako je $\omega_1+\omega_4=2\cos{72^{\circ}}$, pa vrijedi:

\begin{equation*}

\end{equation*}

\noindent\newline Dakle, kako imamo samo korijene i operacije zbrajanja i množenja, dužinu $\overline{ON}$ (dakle, upravo ovaj kosinus) možemo konstruirati. \end{equation*}

\noindent\newline Nadalje, želimo odrediti i $\omega_2$ i $\omega_3$. {\bf Svojstva pravilnog peterokuta}
\end{center}

Ostalo je još prokomentirati odnos duljine stranice peterokuta $a$ i polumjera njemu opisane kružnice $r$. Primijetimo kako za karakteristièni trokut pravilnog peterokuta po kosinusovom pouèku vrijedi:

\begin{equation*}

\end{equation*}

\noindent\newline Iz toga možemo zakljuèiti kako je:

\begin{equation*}
a=r\sqrt{2-\frac{\sqrt{5}-1}{2}}=r\sqrt{\frac{5-\sqrt{5}}{2}}.
\end{equation*}

\noindent\newline Visina $v$ karakteristiènog trokuta na stranicu $a$ je tada jednaka:

\begin{equation*}
v=\sqrt{r^2-r^2\frac{5-\sqrt{5}}{8}}=\frac{r}{2}\sqrt{\frac{\sqrt{5}+3}{2}}.
\end{equation*}

\noindent\newline Tada je površina karakteristiènog trokuta jednaka:

\begin{equation*}
P'=\frac{r^2}{4}\sqrt{\frac{5\sqrt{5}+15-5-3\sqrt{5}}{4}}=\frac{r^2}{4}\sqrt{\frac{5+\sqrt{5}}{2}}.
\end{equation*}

\noindent\newline Površina pravilnog peterokuta je tada jednaka peterostrukoj površini karakteristiènog trokuta, tj. $P=5P'$ ili preciznije:

\begin{equation*}
P=r^2\frac{5}{4}\sqrt{\frac{5+\sqrt{5}}{2}}
\end{equation*}

\noindent\newline Kako je visina karakteristiènog trokuta jednaka $v$, tada je "velika visina", odnosno visina spuštena iz vrha peterokuta na nasuprotnu stranicu jednostavno $w=r+v$, a to je:

\begin{equation*}

\end{equation*}

\noindent\newline Duljine svih dijagonala su jednakih velièina, što je lako dokazivo pomoæu npr. SSS pouèka o sukladnosti dvaju trokuta. Njihova duljina može se izraèunati pomoæu kosinusovog pouèka:

\begin{equation*}

\end{equation*}

\noindent\newline Za to nam je potreban kosinus kuta od $108^{\circ}$. Znamo kosinus kuta od $72^{\circ}$ i znamo kako je $108^{\circ}=180^{\circ}-72^{\circ}$. Dakle, tražimo:

\begin{equation*}
\cos{\left(180^{\circ}-72^{\circ}\right)}=\cos{180^{\circ}}\cos{72^{\circ}}+\sin{180^{\circ}}\sin{72^{\circ}}=-\cos{72^{\circ}}=\frac{1-\sqrt{5}}{4}.
\end{equation*}

\noindent\newline Uvrštavajuæi to u formulu za $d$ imamo:

\begin{equation*}
d=a\sqrt{\frac{3+\sqrt{5}}{2}}=r\sqrt{\frac{5+\sqrt{5}}{2}}.
\end{equation*}

\noindent\newline Definirajmo vektore koji èine stranice peterokuta (uzmimo da $A\in\R^2$ odgovara $\omega_0\in\C$, $B\in\R^2$ odgovara $\omega_1\in\C$, itd.). Tada je:

\begin{eqnarray*}
\overrightarrow{AB}&=&\frac{\sqrt{5}-5}{4}\vec{i}+\frac{\vec{j}}{2}\sqrt{\frac{5+\sqrt{5}}{2}},\\
\overrightarrow{BC}&=&-\frac{\sqrt{5}}{2}\vec{i}+\frac{\vec{j}}{2}\left(\sqrt{\frac{5-\sqrt{5}}{2}}-\sqrt{\frac{5+\sqrt{5}}{2}}\right),
\end{eqnarray*}

\begin{eqnarray*}
\overrightarrow{CD}&=&-\vec{j}\sqrt{\frac{5-\sqrt{5}}{2}},\\
\overrightarrow{DE}&=&\frac{\sqrt{5}}{2}\vec{i}+\frac{\vec{j}}{2}\left(\sqrt{\frac{5-\sqrt{5}}{2}}-\sqrt{\frac{5+\sqrt{5}}{2}}\right),\\
\overrightarrow{EA}&=&\frac{5-\sqrt{5}}{4}\vec{i}+\frac{\vec{j}}{2}\sqrt{\frac{5+\sqrt{5}}{2}}.
\end{eqnarray*}

\noindent\newline Napomenimo samo, prije nego pokažemo zašto su ovi vektori bitni u konstrukciji dodekaedra, kako unutrašnji kut pravilan peterokuta iznosi $72^\circ$. Kako se pravilan peterokut sastoji od pet karakteristiènih trokuta, preostala dva kuta su $\frac{180^{\circ}-72^{\circ}}{2}=54^{\circ}$. Kako svaki krak karakteristiènog trokuta dijeli kut peterokuta na dva jednaka dijela, kut pravilnog peterokuta je jednak $108^{\circ}$ (što se može pokazati i pomoæu $\frac{540^{\circ}}{5}$).

\end{center}

\vskip 0.5cm

\noindent\newline Iduæi skalarni produkt isto nije teško izraèunati buduæi da je jedna komponenta jednaka nuli:

\begin{equation*}

\end{equation*}

\noindent\newline Tada vrijedi:

\begin{equation*}

\end{equation*}

\noindent\newline Vektor ${\bf w}$ sada iznosi:

\begin{equation*}

\noindent\newline No, problem je u tome što je $\sqrt{5}<5$ pa vektor ${\bf u}$ ima negativnu aplikatu. Želimo vršiti rotaciju toèke $C'$ tako da aplikata rotacije bude pozitivna. Takoðer, ovaj vektor nam za formulu treba biti normiran, a obzirom da ima samo aplikatu, bit æe:

\noindent\newline Kako rotacija neæe ovisiti o smjeru vektora $\overrightarrow{AB}$, njegov predznak u vektorskom produktu možemo promijeniti (i onda imati $\overrightarrow{BA}$) i tako dobiti $-{\bf u}$ a time i pozitivnu rotaciju. \end{equation*}

\noindent\newline Uvrštavanjem konkretnih vrijednosti dobivamo:

\begin{eqnarray*}

\noindent\newline Nakon malo sreðivanja, to je:

\noindent\newline Takoðer želimo dobiti formulu za rotaciju toèke $A''$ oko pravca $t\rightarrow B+t\overrightarrow{BC}$ (opet æemo morati uzeti $\overrightarrow{CB}$ kako bismo dobili pozitivan smjer rotacije pri rastu aplikate). Za poèetak imamo:

\end{equation*}

\noindent\newline Norma vektora ${\bf w}$ iznosi:
