\begin{center}
{\bf Konstrukcija skupa $\Z$}
\end{center}

\vskip 0.5cm

Skup $\N=\{1,2,3,\ldots\}$ nazivamo skupom prirodnih brojeva. Skup prirodnih brojeva uz zbrajanje je komutativna polugrupa (zbrajanje je definirano i jedinstveno za svaki $a,b\in\N$, zatim $a+b\in\N$, za svaki $a,b\in\N$; takoðer vrijedi i asocijativnost). Za sada neæemo ulaziti u detaljnija obrazloženja, no pretpostavit æemo, ukoliko ne prihvaæamo ova svojstva sama po sebi oèitima, da je $\N$ definiran pomoæu Peanovih aksioma. Tada se iz njih može formalno definirati zbrajanje i slièno. Takoðer, $\N$ uz $\leq$ je potpuno ureðen skup. Pretpostavit æemo kako za svaki $a,b,c\in\N$, gdje je $a>c$ (ili $b>c$), iz $a=b$ slijedi $a-c=b-c$ i obratno. Takoðer pretpostavit æemo kako je $(a+b)-c=a+(b-c)$, za svaki $a,b,c\in\N$ i $b>c$. Takoðer, neka je $a=a+(b-b)$ i $a=(a-b)+b$, za svaki $a,b\in\N$. Detaljnija pojašnjenja ostavljam za jedan drugi put.

\noindent\newline{\bf Lema.} Neka su $a,b,c,d\in\N$ i neka vrijedi $a+d=b+c$. Neka je $a>b$. Tada je $c>d$.

\noindent\newline{\bf Dokaz.} Kako je $a>b$, tada je i $a+d>b$ pa vrijedi $(a+d)-b=(b+c)-b$. Kako je zbrajanje komutativno, to je $(a+d)-b=(c+b)-b$. Po pretpostavci imamo $(a+d)-b=c+(b-b)$, a to je $(a+d)-b=c$. Opet, po komutativnosti zbrajanja, to je ekvivalentno $(d+a)-b=c$. Po prethodnoj pretpostavci slijedi $d+(a-b)=c$, iz èega imamo $c>d$.

\begin{flushright}
$\square$\\
\end{flushright}

\noindent\newline{\bf Lema.} Neka su $a,b,c,d\in\N$ i neka vrijedi $a+d=b+c$. Neka je $a=b$. Tada je $c=d$.

\noindent\newline{\bf Dokaz.} Imamo $a+d=b+c$. Kako je $a=b$ to je $a+d=a+c$. Iz toga slijedi, jer je $(a+d)>a$ i $(a+c)>c$, da je $(a+d)-a=(a+c)-a$. Po komutativnosti imamo $(d+a)-a=(c+a)-a$, tj. $d+(a-a)=c+(a-a)$ i napokon $d=c$, tj. $c=d$.

\begin{flushright}
$\square$\\
\end{flushright}

\noindent{\bf Definicija.} Neka je za svaki $(a,b),(c,d)\in\N\times\N$ definirana relacija $(a,b)\sim(c,d)$ ako i samo ako $a+d=b+c$.

\noindent\newline{\bf Propozicija.} Relacija $\sim$ iz prethodne definicije je relacija ekvivalencije.

\noindent\newline{\bf Dokaz.} {\it Refleksivnost.} Vrijedi $a+b=a+b$ pa je $(a,b)\sim(a,b)$. {\it Simetriènost.} Neka je $(a,b)\sim(c,d)$. Tada vrijedi $a+d=b+c$, što je ekvivalentno $b+c=a+d$. Kako je zbrajanje komutativno, prethodan izraz ekvivalentan je $c+b=d+a$, što povlaèi $(c,d)\sim(a,b)$. {\it Tranzitivnost.} Neka je $(a,b)\sim(c,d)$ i $(c,d)\sim(e,f)$. To znaèi $a+d=b+c$ i $c+f=d+e$. Pretpostavimo kako je $a>b$. Tada po prethodnoj lemi imamo $c>d$. To opet povlaèi $e>f$. Stoga, iz $a+d=b+c$ slijedi $d+a=c+b$, zatim $(d+a)-b=(c+b)-b$ pa $d+(a-b)=c+(b-b)$ i napokon $c=(a-b)+d$. Iz $c+f=d+e$ slijedi $(c+f)-f=(d+e)-f$ te $c+(f-f)=d+(e-f)$, što je $c=(e-f)+d$. Dakle, $(a-b)+d=(e-f)+d$, tj. $a-b=e-f$. Iz toga imamo $(a-b)+b=(e-f)+b$, a to je $a=(e-f)+b$, tj. $a=b+(e-f)$. Zatim iz toga slijedi $a+f=(b+(e-f))+f$, što je po asocijativnosti $a+f=b+((e-f)+f)$. Iz toga napokon dobivamo $a+f=b+e$ što povlaèi $(a,b)\sim(e,f)$. Dokaz za $a<b$ se provodi analogno. Ukoliko je $a=b$, po prethodnoj lemi mora biti $c=d$, a zatim i $e=f$. Stoga je $a+e=b+e$ pa $a+f=b+e$, pa je opet $(a,b)\sim(e,f)$.

\begin{flushright}
$\square$\\
\end{flushright}

\noindent{\bf Definicija.} Neka je $[(a,b)]=\{(c,d)\in\N\times\N:\ (a,b)\sim (c,d)\}$. Neka su $a,m\in\N$. Definiramo:

\begin{eqnarray*}
m^{+}:&=&[(a+m,a)],\\
0:&=&[(a,a)],\\
m^{-}:&=&[(a,a+m)].
\end{eqnarray*}

\noindent\newline Nadalje, neka je $\Z:=\{m^{+}:\ m\in\N\}\cup\{0\}\cup\{m^{-}:\ m\in\N\}$.

\noindent\newline{\bf Propozicija.} Neka su $a,m\in\N$ i $\pi:\N\rightarrow\N$ funkcija definirana s $\pi(x)=x+1$. Tada je $[(a,a+m)]=[(1,\pi(m))]$, $[(a,a)]=[(1,1)]$ i $[(a+m,a)]=[(\pi(m),1)]$.

\noindent\newline{\bf Dokaz.} Direktno slijedi iz svojstva $x\in[y]$ povlaèi $[x]=[y]$. Imamo $(1,\pi(m))\sim(a,a+m)$ jer $1+(a+m)=\pi(m)+a$, tj. $1+(m+a)=\pi(m)+a$ što nas dovodi do $(1+m)+a=\pi(m)+a$; to je ekvivalentno $(m+1)+a=\pi(m)+a$, tj. $\pi(m)+a=\pi(m)+a$. Stoga, $(1,\pi(m))\in[(a,a+m)]$ pa je $[(a,a+m)]=[(1,1+m)]$. Ostali se dokazi provode analogno.

\begin{flushright}
$\square$\\
\end{flushright}

\noindent{\bf Definicija.} Neka su $(x,y),(z,w)\in\N\times\N$. Definiramo binarnu operaciju $+:\Z\rightarrow\Z$ kao $[(x,y)]+[(z,w)]:=[(x+z,y+w)]$.

\noindent\newline{\bf Propozicija.} Skup $\Z$ uz operaciju iz prethodne definicije je komutativna grupa.

\noindent\newline{\bf Dokaz.} Zatvorenost i definiranost (uz nasljedstvo definiranosti zbrajanja iz $\N$) vrijede po definiciji; kako je $(x+z,y+w)\in\N\times\N$, imamo $[X]\in\Z$ takav da je $(x+z,y+w)\in[X]$ što povlaèi $[X]=[(x+z,y+w)]$. Zatim, operacija je jedinstveno definirana, jer, uzmemo li $(x_1,y_1),(x_2,y_2),(z_1,w_1),(z_2,w_2)\in\N\times\N$ takve da je $(x_1,y_1)=(x_2,y_2)$ i $(z_1,w_1)=(z_2,w_2)$, imamo $[(x_1,y_1)]+[(z_1,w_1)]=[(x_1+z_1,y_1+w_1)]$ te $[(x_2,y_2)]+[(z_2,w_2)]=[(x_2+z_2,y_2+w_2)]$. No, kako je $(x_1,y_1)=(x_2,y_2)$ i $(z_1,w_1)=(z_2,w_2)$, takoðer je i $[(x_1,y_1)]=[(x_2,y_2)]$ te $[(z_1,w_1)]=[(z_2,w_2)]$ (po obiènoj supstituciji). Tada imamo $[(x_1,y_1)]+[(z_1,w_1)]=[(x_2+z_2,y_2+w_2)]$, tj. $[(x_1+z_1,y_2+w_2)]=[(x_2+z_2,y_2+w_2)]$. {\it Asocijativnost.} Imamo $\left([(x,y)]+[z,w]\right)+[(p,q)]=[(x+z,y+w)]+[(p,q)]=[((x+z)+p,(y+w)+q)]$. Obzirom da vrijedi asocijativnost u $\N$, imamo $[((x+z)+p,(y+w)+q)]=[(x+(z+p),y+(w+q))]=[(x,y)]+[(z+p,w+q)]=[(x+y)]+\left([(z,w)]+[(p,q)]\right)$. {\it Neutralan element.} Imamo $[(x,y)]+[(1,1)]=[(x+1,y+1)]$. No, kako je $x+(y+1)=y+(x+1)$, vrijedi $(x,y)\sim(x+1,y+1)$ što implicira $[(x,y)]=[(x+1,y+1)]$. Dakle, $[(x,y)]+[(1,1)]=[(x,y)]$. Isto se pokaže i za $[(1,1)]+[(x,y)]=[(x,y)]$. Stoga je $[(1,1)]\in\Z$ neutralan element. {\it Inverzni elementi.} Neka je $[(x,x+m)]\in\Z$. Tada je $[(x+m,x)]\in\Z$ i imamo $[(x,x+m)]+[(x+m,x)]=[(x+m,x+m)]=[(1,1)]$ (po prethodnoj propoziciji). Slièno se pokaže i za $[(x,x+m)]\in\Z$, te za $[(x,x)]\in\Z$. {\it Komutativnost.} Neka je $[(x,y)],[(z,w)]\in\Z$. Tada je $[(x,y)]+[(z,w)]=[(x+z,y+w)]=[(z+x,w+y)]=[(z,w)]+[(x,y)]$.

\begin{flushright}
$\square$\\
\end{flushright}

\noindent\newline {\bf Primjedba.} Primijetimo kako iz gornje propozicije slijedi da je $0\in\Z$ neutralan element obzirom na operaciju zbrajanja u $\Z$. Takoðer, uzmemo li $m^{+}\in\Z$, njegov inverzan element obzirom na zbrajanje je $m^{-}\in\Z$, tj. vrijedi $m^{+}+m^{-}=m^{-}+m^{+}=0$.

\noindent\newline{\bf Definicija.} Neka je $\leq$ relacija na skupu $\Z$ takva da $[(x,y)]\leq[(z,w)]$ ako i samo ako $x+w\leq y+z$, za svaki $[(x,y)],[(z,w)]\in\Z$.

\noindent\newline{\bf Propozicija.} Skup $\Z$ uz $\leq$ je potpuno ureðen skup.

\noindent\newline{\bf Dokaz.} Po definiciji relacije, relacija je definirana za svaka dva elementa skupa $\Z$. {\it Refleksivnost.} Vrijedi $x+y\leq x+y$ pa je $[(x,y)]\leq[(x,y)]$. {\it Antisimetriènost.} Neka je $[(x,y)]\leq[(z,w)]$ i $[(z,w)]\leq[(x,y)]$. Tada je $x+w\leq y+z$ i $z+y\leq w+x$, iz èega slijedi $x+w=y+z$, tj. $[(x,y)]\sim[(z,w)]$ pa onda $[(x,y)]=[(z,w)]$. {\it Tranzitivnost.} $[(x,y)]\leq[(z,w)]$ i $[(z,w)]\leq[(p,q)]$ je ekvivalentno $x+w\leq y+z$ i $z+q\leq w+p$. Vrijedi $w\leq y+z-x$ i $z+q-p\leq w$. Stoga imamo $z+q-p\leq y+z-x$ i $q+x\leq y+p$, tj. $x+q\leq y+p$ što povlaèi $[(x,y)]\leq[(p,q)]$.

\begin{flushright}
$\square$\\
\end{flushright}

\noindent\newline{\bf Primjedba.} Slièno se definira slabija relacija $<$ uz koju je skup $\Z$ parcijalno ureðen skup. Dakle, $[(x,y)]<[(z,w)]$ ako i samo ako $x+w<y+z$, za svaki $[(x,y)],[(z,w)]\in\Z$.

\noindent{\bf Propozicija.} Vrijedi $m^{-} < 0 < n^{+}$, za svaki $m^{-},n^{+}\in\Z$.

\noindent\newline{\bf Dokaz.} Imamo $m^{-}=[(1,\pi(m))]$, $0=[(1,1)]$ i $n^{+}=[(\pi(m),1)]$. Vrijedi $1+1<\pi(m)+1$, tj. $1<\pi(m)$, što je istinito za svaki $m\in\N$ (jer broj $1$ nije sljedbenik nijednom prirodnom broju). Dakle, $m^{-}<0$. Takoðer, $1+1<\pi(m)+\pi(n)$, tj. $2<\pi(m)+\pi(n)$. Obzirom da je $1<\pi(m)$ i $1<\pi(n)$ oèito je $2<\pi(m)+\pi(n)$ pa je $m^{-}<n^{+}$, za svaki $m^{-},n^{+}\in\Z$. Takoðer, $1+1<1+\pi(m)$, tj. $1<\pi(m)$ pa je i $0<n^{+}$.

\begin{flushright}
$\square$\\
\end{flushright}

\noindent{\bf Napomena.} Uvedimo sada oznaku $m:=m^{+}$ i $-m:=m^{-}$.

\noindent\newline{\bf Propozicija.} Neka je $m\in\Z$. Ako je $m<0$ tada je $-m>0$ (u smislu aditivnog inverza). Takoðer, ako je $m>0$ tada je $-m<0$.

\noindent\newline{\bf Dokaz.} Pretpostavimo da je $m<0$. Tada je, po prethodnoj propoziciji, $m=[(1,\pi(m))]$, $0=[(1,1)]$ pa zbog $m<0$ vrijedi $1+1<\pi(m)+1$, tj. $1<\pi(m)$. Pretpostavimo kako je $-m\leq 0$. Tada bi, zbog $-m=[(\pi(m),1)]$ vrijedilo $\pi(m)+1\leq 1+1$, tj. $\pi(m)\leq 1$, što je u suprotnosti s $\pi(m)>1$. Dakle, $m<0$ povlaèi $-m>0$. Po rezultatu iz teorije grupa, kako je $-m$ aditivni inverz od $m$, vrijedi $-(-m)=m$. Dokaz za $m>0$ se provodi analogno.

\begin{flushright}
$\square$\\
\end{flushright}

\noindent{\bf Propozicija.} Neka su $m,n\in\Z$. Ako je $m\leq n$, tada $-n\leq -m$ (u smislu aditivnih inverza).

\noindent\newline{\bf Dokaz.} Pretpostavimo kako je $m>0$ i $n>0$. Tada je $m=[(\pi(m),1)]$ i $n=[(\pi(n),1)]$. Tada je $-m=[(1,\pi(m))]$ i $-n=[(1,\pi(n))]$. Iz $[(\pi(m),1)]\leq[(\pi(n),1)]$ slijedi $\pi(m)+1\leq \pi(n)+1$, tj. $\pi(m)\leq\pi(n)$. Pretpostavimo da je $-m<-n$. Tada vrijedi $1+\pi(n)<1+\pi(m)$, tj. $\pi(n)<\pi(m)$. No, to je u suprotnosti s pretpostavkom $\pi(m)\leq\pi(n)$. Dokaz za $m<0$ i $n<0$ slijedi analogno. Sluèaj kada je $m>0$ i $n<0$ uz $m\leq n$ je nemoguæ po prethodnoj propoziciji. Pretpostavimo kako je $-n<-m$. No, tada je $-n>0$ i $-m<0$. No, uz $-n<-m$ to je nemoguæe pa može biti samo $-m\leq -n$.

\begin{flushright}
$\square$\\
\end{flushright}

\noindent{\bf Teorem (Well Ordering of $\Z^{+}$).} Neka je $A\subseteq\Z^{+}$ i $A\neq\emptyset$. Tada postoji $m\in A$ takav da je $m\leq a$ za svaki $a\in A$.

\noindent\newline{\bf Dokaz.} Dokaz se provodi matemematièkom indukcijom. Prvo pokažimo kako tvrdnja vrijedi za sve konaène podskupove od $\Z^{+}$. Neka je $|A_1|=1$ i $A_1\subseteq\Z^{+}$, i.e. $A_1=\{a\}$. Tada je oèito $\min{A_1}=a$. Pretpostavimo kako tvrdnja vrijedi za neki $k\in\N$; neka za svaki skup $A_k\subseteq\Z^{+}$, gdje je $|A_k|=k$, postoji $m\in A$ takav da je $m\leq a$, za svaki $a\in A_k$. Neka je $A_{k+1}\subseteq\Z^{+}$ takav da je $|A_{k+1}|=k+1$. Neka je $a'\in A_{k+1}$. Tada je $\left|A-\{a'\}\right|=k$ pa postoji $m\in A-\{a'\}$ takav da je $m\leq a$, za svaki $a\in A-\{a'\}$. Vrijedi $m\leq a'$ ili $m\leq a'$. U prvom sluèaju je oèito $\min{A_{k+1}}=m$ (jer je tada i dalje $m\leq a$, za svaki $a\in A_{k+1}$), a u drugom sluèaju je $\min{A_{k+1}}=a'$ (jer je tada $a'\leq m\leq a$, za svaki $a\in A_{k+1}$). Stoga tvrdnja vrijedi za sve skupove velièine $n\in\N$.

Pokažimo sada kako tvrdnja vrijedi i za beskonaène podskupove od $\Z^{+}$. Neka je $A\in\Z^{+}$. Neka je $A_x=\{0,\ldots,x\}$, za $x\in\Z^{+}$. Tada je $A\cap A_x$ konaèan skup i vrijedi kako postoji $m\in A\cap A_x$ takav da je $m\leq a$, za svaki $a\in A\cap A_x$. Po definiciji presjeka, ako je $a\in A\cap A_x$ tada je $a\in A$ i $a\cap A_x$. Stoga, slabljenjem tvrdnje, postoji $m\in A$ takav da je $m\leq a$, za svaki $a\in A$.

\begin{flushright}
$\square$\\
\end{flushright}

\noindent\newline{\bf Definicija.} Neka je $x\in\R$. Tada funkciju $|\cdot|:\R\rightarrow\R^{+}_0$ definiranu formulom

\begin{equation*}
|x|=x\mathcal{I}_{\left[0,\infty\right]}(x)+(-x)\mathcal{I}_{\left\langle-\infty,0\right\rangle}(x),
\end{equation*}

\noindent\newline gdje je $\mathcal{I}_S:\R\rightarrow\{0,1\}$ indikator funkcija za skup $S$, nazivamo apsolutna vrijednost (broja $x$).

\noindent\newline{\bf Primjedba.} Iz definicije je lako vidjeti kako vrijedi $|x|\geq 0$, za svaki $x\in\R$.

\noindent\newline{\bf Propozicija.} Za svaki $a\in\R$ vrijedi $a\leq|a|$.

\noindent\newline{\bf Dokaz.} Ako je $a\geq 0$ tada je $|a|=a$, po definiciji. Ako je $a<0$ tada je $|a|=-a$ i vrijedi $|a|>0$ pa tako i $|a|>a$ (jer je $a<0$). Uzevši oba sluèaja u obzir to je $|a|\geq a$.

\begin{flushright}
$\square$\\
\end{flushright}

\noindent{\bf Propozicija.} Neka su $a,b\in\R$. Ako je $a=b$ tada je $|a|=|b|$. Obrat opæenito ne vrijedi.

\noindent\newline{\bf Dokaz.} Neka je $a=b$. Uzmimo prvo $a\geq0$. Tada je i $b\geq0$ pa je $|a|=a$ i $|b|=b$. Iz pretpostavke propozicije direktno slijedi $|a|=a=b=|b|$, tj. $|a|=|b|$ (ako su $a,b\in\R^{+}_0$). Dalje, uzmimo $a<0$. Tada je i $b<0$ pa je $|a|=-a$ i $|b|=-b$. Vrijedi $-(-a)=a$ i $-(-b)=b$. Iz pretpostavke je $-|a|=-(-a)=a=b=-(-b)=-|b|$, i.e. $-|a|=-|b|$. To je $|a|=|b|$ za $a<0$ i $b<0$ pa uzevši i prvi sluèaj u obzir vrijedi $|a|=|b|$, za svaki $a,b\in\R$.

\begin{flushright}
$\square$\\
\end{flushright}

\noindent{\bf Propozicija.} Neka su $a,b\in\Z$. Vrijedi $|a|\cdot |b|=|a\cdot b|$.

\noindent\newline{\bf Dokaz.} Neka je $a\geq 0$ i $b\geq0$. Tada je $a b>0$ i stoga $|a b|=a b$. Isto tako je i $|a|=a$ i $|b|=b$ pa je $|a||b|=a b$. Iz oba izraza slijedi $|a b|=|a||b|$ za $a,b\geq 0$. Ukoliko su $a,b<0$ vrijedi $a b>0$ pa je $|a b|=a b$. Isto tako je $|a|=-a$ i $|b|=-b$ pa je $|a||b|=-a\cdot(-b)=a b=|a b|$. Ako je $a\geq 0$ i $b<0$ (isto se pokaže i za $a<0$ i $b\geq 0$), vrijedi $a b<0$ pa je $|a b|=-a b$. No, isto tako $|a|=a$ i $|b|=-b$ pa je $|a||b|=a\cdot(-b)=-a b=|a b|$. Time smo iscrpili sve moguænosti i vrijedi tvrdnja propozicije.

\begin{flushright}
$\square$\\

\noindent\newline{\bf Definicija.} Neka je $a\in\Z$ i $b\in\Z\backslash\{0\}$. \noindent\newline{\bf Primjer.} Pogledajmo par primjera kako bi nam bilo jasnije. Broj $6$ je djeljiv s $3$ jer vrijedi $6=3\cdot 2$ (ovdje je $k=2$ i vrijedi $3|6$). No što ako imamo $-6$ i $3$? \noindent\newline{\bf Propozicija.} Ako je $b\in\Z\backslash\{0\}$ djelitelj od $a\in\Z$ vrijedi $b\leq|b|\leq|a|$.

\noindent\newline{\bf Dokaz.} Po prethodnoj propoziciji, ako je $a=k b$ tada je i $|a|=|k b|=|k||b|$. Kako su $|a|$, $|k|$ i $|b|$ nenegativni (što je lako uoèiti iz definicije apsolutne vrijednosti), tada je $|b|\leq|a|$. Po prethodnoj propoziciji je $b\leq|b|$ pa je $b\leq|b|\leq|a|$.

\noindent{\bf Teorem (o dijeljenju s ostatkom).} Za svaki $a,b\in\Z$, $b\neq 0$, postoje jedinstveni $q\in\Z$ ({\bf kvocijent}) i $r\in\N_0$ ({\bf ostatak}) takvi da vrijedi $a=bq+r$ i $0\leq r<|b|$.

\noindent{\bf Definicija.} Neka su zadani $a,b\in\Z\backslash\{0\}$. Prirodan broj $x\in\N$ za koji vrijedi da $x|a$ i $x|b$ zovemo {\bf zajednièki djelitelj} od $a$ i $b$. \noindent\newline{\bf Definicija.} Brojeve $a,b\in\Z\backslash\{0\}$ za koje vrijedi $\gcd(a,b)=1$ kažemo da su {\bf relativno prosti}. Za broj $p\in\N\backslash\{1\}$ kažemo da je {\bf prost} ukoliko ne postoji $n\in\N\backslash\{1,p\}$ takav da $n|p$. U suprotnom kažemo da je {\bf složen}. 

\noindent\newline{\bf Propozicija.} Neka su $a,b,c\in\Z$, takvi da $c|a$ i $c|b$. Tada $c|(a x+b y)$, za svaki $x,y\in\Z$.

\noindent\newline{\bf Dokaz.} Kako $c|a$ i $c|b$, postoje $k_1,k_2\in\Z$ takvi da vrijedi $a=c k_1$ i $b=c k_2$. Neka su $x,y\in\Z$. Pokažimo da postoji $k\in\Z$ takav da je $a x+b y=c k$. Pronašli smo $k_1 x+k_2 y=k\in\Z$ takav da vrijedi $w=c k$, tj. $a x+b y=c k$ pa $c|(a x+b y)$.

$\square$\\
\end{flushright}

\noindent{\bf Propozicija.} Neka su $a,b,q,r\in\Z$ te $a,b,r\neq0$, $0<r<|b|$ takvi da vrijedi $a=b q+r$. Tada je $\gcd(a,b)=\gcd(b,r)$.

\noindent{\bf Teorem (Bezoutova lema).} Neka su $a,b\in\Z\backslash\{0\}$. Vrijedi:

\begin{equation*}
\gcd(a,b)=\min{\{a x+b y:\ x,y\in\Z\}}\cap\N.
\end{equation*}

\noindent\newline{\bf Dokaz.} Neka je $L=\{a x+b y:\ x,y\in\Z\}\cap\N$. Uzmimo $g_1=\gcd(a,b)$ i $g_2=\min{L}$. Pokažimo da je $g_1=g_2$. No, prvo moramo pokazati da $g_2$ postoji. Ako su $a,b>0$, dovoljno je uzeti $x=1$ i $y=1$ jer æe tada biti $a+b>0$, a time i $(a+b)\in L$. Ako je $a,b<0$, uzimamo $x=y=-1$ i tada je $-a-b>0$ i $-a-b\in L$. Ako je $a>0$ i $b<0$ (ili obratno) možemo uzeti $x=1$ i $y=-1$ pa æe biti $a-b>0$ i $a-b\in L$ (ako je obratno, tj. $a<0$ i $b>0$ tada je $b-a\in L$). Dakle, skup $L$ je neprazan i sadrži samo pozitivne brojeve pa po {\it well-ordering} principu, ima i najmanji element $g_2=\min{L}$.

Pretpostavimo da $g_2$ ne dijeli $a$ (ili $g_2$ ne dijeli $b$). Tada postoje jedinstveni $q,r\in\Z$ takvi da vrijedi $a=g_2 q+r$, gdje je $0<r<g_2$ (ako bi bilo $0\leq r$, uzeli bismo u obzir i to da je $a$ djeljiv s $g_2$). To znaèi da je $a=a q x+b q y+r$. Imamo $a-a q x-b q y=r$. Izluèimo faktore tako da bude $a(1-q x)+b(-q y)=r$. Kako su $1-q x,-q y\in\Z$ vrijedi da je $r=a(1-q x)+b(-q y)\in L$. No, kako je $0<r<g_2=a x+b y$, tj. $0<a(1-q x)+b(-q y)<a x-b y$, došli smo u kontradikciju s pretpostavkom da je $a x-b y$ najmanji element u $L$. Dakle, $g_2|a$ i $g_2|b$.

Promotrimo sada $g_1=\gcd(a,b)$. Kako $g_2|a$ i $g_2|b$ oboje su zajednièki djelitelji od $a$ i $b$. No, $g_1$ je najveæi zajednièki djelitelj pa vrijedi $g_1\geq g_2$. Nakon izluèivanja to je $g_2=g_1(k_1 x+k_2 y)$, dakle $g_1|g_2$ te je $g_2\geq g_1$. Kako imamo $g_2\geq g_1$ i $g_1\geq g_2$, slijedi da je $g_1=g_2$, tj. $\gcd(a,b)=\min{\left(\{a x+b y:\ x,y\in\Z\}\cap\N\right)}$.

\begin{flushright}
$\square$\\
\end{flushright}

\noindent{\bf Teorem (Euklidova lema).} Neka su zadani brojevi $a,b,c\in\Z$ takvi da vrijedi $\gcd(a,b)=1$ i $a|b c$. Tada $a|c$.

$\square$\\
\end{flushright}

\noindent{\bf Lema.} Neka su $a,b\in\Z\backslash\{0\}$ takvi da $a|b$. Tada je $\gcd{(a,b)}=|a|$. Vrijedi i obrat.

\noindent\newline{\bf Dokaz.} {\it Nužnost.} Znamo kako $a|b$, ali i da $a|a$. To je najveæi broj koji dijeli $a$ i vrijedi $a|b$ i $a|a$ pa je $\gcd(a,b)=|a|$ (dodajemo apsolutnu vrijednost kako bismo osigurali da je $\gcd(a,b)\in\N$).

{\it Dovoljnost.} Pretpostavimo kako je $\gcd{(a,b)}=|a|$. To znaèi kako $|a|$ dijeli $a$ i $|a|$ dijeli $b$, tj. postoji $k\in\Z$ takav da je $b=|a| k$. Ukoliko je $b$ negativan, tada mora biti da je $k$ negativan. Stoga, ako je $a$ negativan, uzimamo $b=a k'$, gdje je $k'\in\Z$ i $k'=|k|$ pa vrijedi $a|b$. Ukoliko je $a$ pozitivan, jednostavno imamo $b=a k$ i vrijedi $a|b$. Ako je $b$ pozitivan, mora biti i da je $k$ pozitivan pa za pozitivan $a$ uzimamo $b=a k$ i onda $a|b$. Ako je $a$ pak negativan, dovoljno je uzeti $k'=-k$ i imamo $b=a k'$, gdje je $k'\in\Z$ pa opet $a|b$. Kako smo u svim sluèajevima dobili $a|b$ vrijedi pretpostavka obrata tvrdnje.

\begin{flushright}
$\square$\\
\end{flushright}

\noindent{\bf Teorem (Euklidov algoritam).} Neka su zadani $a,b\in\Z\backslash\{0\}$. Tada, postupkom

\begin{eqnarray*}
a&=&b q_0+r_1,\\
b&=&r_1 q_1+r_2,\\
r_1&=&r_2 q_2+r_3,\\
\vdots\\
r_{n-2}&=&r_{n-1} q_{n-1}+r_n,\\
r_{n-1}&=&r_n q_n,
\end{eqnarray*}

\noindent\newline gdje je $0<r_i<r_{i-1}$, za $i\in\{2,\ldots,n\}$ te $0<r_1<|a|$, dobivamo $\gcd(a,b)=r_{n}$.

\noindent\newline{\bf Dokaz.} Ovaj postupak traje konaèno upravo zbog uzastopne primjene teorema o dijeljenju s ostatkom. Obzirom da je $0<r_1<|a|$ i $0<r_i<r_{i-1}$, za $i\in\{2,\ldots,n\}$ zbog {\it well-ordering} svojstva skupa $\Z^{+}$, moramo doæi do nule (što se vidi u $r_{n-1}=r_n q_n$). Zatim, po prethodnoj propoziciji vrijedi $\gcd(a,b)=\gcd(b,r_1)$. No, tako vrijedi i, po drugom koraku, $\gcd(b,r_1)=\gcd(r_1,r_2)$. Tako za $n$-ti ostatak vrijedi $\gcd(r_{n-2},r_{n-1})=\gcd(r_{n-1},r_n)$. Tako je $\gcd(a,b)=\gcd(r_{n-1},r_n)$. No, kako $r_n|r_{n-1}$, po prethodnoj lemi vrijedi $\gcd(r_{n-1},r_n)=r_n$. Tako je $\gcd(a,b)=r_n$.

\begin{flushright}
$\square$\\
\end{flushright}

\noindent{\bf Propozicija.} Neka su $a,b\in\Z^{\ast}$. Iz Euklidovog algoritma možemo dobiti $x,y\in\Z$ takve da je $\gcd{(a,b)}=a x+b y$.

\noindent\newline{\bf Dokaz.} Primijenimo li Euklidov algoritam na $a$ i $b$, dobivamo:

\begin{eqnarray*}
a&=&b q_0+r_1,\\
b&=&r_1 q_1+r_2,\\
r_1&=&r_2 q_2+r_3,\\
\vdots\\
r_{n-3}&=&r_{n-2} q_{n-2}+r_{n-1},\\
r_{n-2}&=&r_{n-1} q_{n-1}+r_n,\\
r_{n-1}&=&r_n q_n.
\end{eqnarray*}

\noindent\newline Uzmemo $r_n=r_{n-2}-r_{n-1} q_{n-1}$. Zatim, $r_n=r_{n-2}-(r_{n-3}-r_{n-2} q_{n-2}) q_{n-1}$, tj. $r_n=r_{n-2}(1+q_{n-2})-r_{n-3} q_{n-1}$. Pretpostavimo kako $r_n=r_{i+1}x'+r_{i}y'$. Tada, jer je $r_{i-1}=r_{i} q_{i}+r_{i+1}$ imamo $r_n=x'(r_{i-1}-r_i q_{i})+r_i y'$, tj. $r_n=x' r_{i-1}+r_i(y'-q_{i} x')$. Dakle, nastavimo li tom analogijom, dolazimo i da $r_n=x' r_2+y' r_1$. No, $r_2=b-r_1 q_1$ pa imamo $r_n=x'(b-r_1 q_1)+y' r_1$, tj. $r_n=x' b+r_1(y'-q_1 x')$. Uzmimo $y'-q_1 x'=x$. Dalje, $r_1=a-b q_0$ pa je $r_n=x' b+(a-b q_0)x$, tj. $r_n=b(x'-q_0 z)+a x$. Uzemo li $y=(x'-q_0 z)$ imamo $r_n=a x+b y$, tj. $\gcd{(a,b)}=a x+b y$ za $x,y\in\Z$.

\begin{flushright}
$\square$\\
\end{flushright}

\noindent\newline{\bf Propozicija.} Neka je $a\in\Z$. Tada je $\gcd(a,0)=a$.

\noindent\newline{\bf Dokaz.} Vrijedi da $a|a$ i $a|0$. To je najveæi takav broj koji dijeli $a$ pa je $\gcd(a,0)=a$.

\noindent{\bf Propozicija.} Za $a,b,k\in\Z$, $a,b\neq 0$, vrijedi $\gcd(a,b)=\gcd(a,b+a k)$.

\noindent\newline{\bf Dokaz.} Uzmimo $g=\gcd(a,b)$. Tada $g|a$ i $g|b$ i to je najveæi takav broj. Tada $g$ dijeli i cjelobrojnu kombinaciju od $a$ i $b$ pa i $b+a k$. Stoga $g|a$ i $g|(b+a k)$. Uzmimo $g'=\gcd(a,b+a k)$. Tada vrijedi $g'\geq g$ jer je $g$ djelitelj od $a$ i $b+a k$, a $g'$ je najveæi takav. No, kako $g'|a$ i $g'|(b+a k)$, tada je $a=g'x$, za neki $x\in\Z$ i $b+a k=g'y$, za neki $y\in\Z$. Tako je $b+g'x k=g'y$, tj. $b=g'y-g'x k=g'(y-x k)$. Dakle, $g'|b$. Kako $g'|a$ i $g'|b$ on je zajednièki djelitelj od $a$ i $b$ pa je $g'\leq g$ jer je $g$ najveæi takav. No, imamo $g'\leq g$ i $g'\geq g$ pa je $g'=g$.

Drugi naèin da se pokaže ova propozicija je ovisan o prethodno dokazanoj propoziciji. Uzmemo li $c=b+a k$, oèito je kako je $b$ ostatak pri dijeljenju $c$ s $a$. Pa ako je $g=\gcd(a,c)$ tada je, po prethodno dokazanoj propoziciji $\gcd(a,c)=\gcd(a,b)$. To je $\gcd(a,b+a k)=\gcd(a,b)$.

\begin{flushright}
$\square$\\
\end{flushright}

\noindent{\bf Propozicija.} Neka su $m,n,k\in\Z^{\ast}$ takvi da $\gcd{(m,n)}=1$ i $k|n$. Tada je $\gcd{(m,k)}=1$.

\noindent\newline{\bf Dokaz.} Kako $k|n$, postoji $q\in\Z$ takav da je $n=q k$. Pretpostavimo kako postoji $r,m',k'\in\Z$ tako da vrijedi $m=r m'$ i $k=r k'$. No kako je $r\in\Z$ i $(x m'+y q k')\in\Z$, oba broja moraju biti jednaka ili $1$ ili $-1$. Dakle $r=\pm 1$, što znaèi da su brojevi $m$ i $k$ relativno prosti.

\begin{flushright}
$\square$\\
\end{flushright}

\noindent{\bf Propozicija.} Neka su $m,n,k\in\Z^{\ast}$ takvi da $\gcd{(m n,k)}=1$. Tada $\gcd{(m,k)}=1$ i $\gcd{(n,k)}=1$.

\noindent\newline{\bf Dokaz.} Imamo $\gcd{(m n,k)}=1$. Neka je $\gcd{(m,k)}=q$. To znaèi kako postoje $u,v\in\Z$ takvi da je $m=q u$ i $k=q v$. Iz toga slijedi kako je $m n=q u n$. No, $(u n w+v z)\in\Z$ pa stoga mora biti i $u n w+v z=1$ i imamo $q=1$. Dakle, jedini zajednièki djelitelj od $m$ i $k$ je $1$ pa mora biti $\gcd{(m,k)}=1$. Analogno se dokazuje i za $\gcd{(n,k)}=1$.

\begin{flushright}
$\square$\\
\end{flushright}

\noindent{\bf Propozicija.} Neka su $k,n,m\in\Z^{\ast}$ takvi da je $\gcd{(n,m)}=1$. Tada je $\gcd{(k n,k m)}=k$.

\noindent\newline{\bf Dokaz.} Neka je $g=\gcd{(k n,k m)}$. Pokažimo da je $g=k$. Oèito je kako $k|(k n)$ i $k|(k m)$, stoga je $k$ zajednièki djelitelj od $k n$ i $k m$ pa $k|g$. Pokažimo da $g|k$. No, po Euklidovoj lemi mora biti da $n|x$, tj. $\frac{x}{n}\in\Z$. Uzmimo $c=\frac{x}{n}$. Stoga iz $k n=g x$ imamo $k=g\frac{x}{n}$, i.e. $k=g c$, za $c\in\Z$ pa po definiciji $g|k$. Stoga je $g=k$.

\begin{flushright}
$\square$\\
\end{flushright}

\noindent{\bf Propozicija.} Neka su $m,n\in\Z^{\ast}$. Tada je $\gcd{(m n,m)}=m$.

\noindent\newline{\bf Dokaz.} Uzmimo $g=\gcd{(m n,m)}$. Pokažimo kako je $g=m$. Oèito $g|(m n)$ i $g|m$ (iz toga slijedi $g\leq m$). No, isto tako $m|(m n)$ i $m|m$ pa je $m$ zajednièki djelitelj od $m n$ i $m$ te vrijedi, obzirom da je $g$ po pretpostavci najveæi, $m\leq g$. Kako imamo $g\leq m$ i $m\leq g$, vrijedi $g=m$.

$\square$\\
\end{flushright}

\noindent{\bf Propozicija.} Neka su $k,l,n\in\Z^{\ast}$ takvi da je $\gcd{(k,n)}=1$ i $\gcd{(l,n)}=1$. Tada je $\gcd{(k l,n)}=1$.

\noindent\newline{\bf Dokaz.} Pretpostavimo da postoji $q\in\Z$ takav da $q|(k l)$ i $q|n$ te $q\neq\pm1$. To znaèi kako je $k l=q x$ i $n=q y$, za $x,y\in\Z$. Stoga je $\frac{y}{n}\in\Z$. Iz $n=q y$ dobivamo $1=q\frac{y}{n}$. No, to znaèi da $q|1$ i $\frac{y}{n}|1$, a to može biti samo ako $q=\frac{y}{n}=\pm1$ (fali preciznosti, ali bit je tu). $\square$\\

\noindent{\bf Napomena.} Od sada æemo u tekstu uzeti oznaku $\Z^{\ast}=\Z\backslash\{0\}$.

\noindent\newline{\bf Definicija.} Neka je $a,b\in\Z^{\ast}$. Svaki broj $m\in\Z^{\ast}$ takav da $m|a$ i $m|b$ zovemo {\bf zajednièki višekratnik} od $a$ i $b$. Najmanji takav broj zovemo {\bf najmanji zajednièki višekratnik} od $a$ i $b$ te pišemo $m=\textnormal{lcm}(a,b)$.

\noindent\newline{\bf Propozicija.} Za svaki $a,b\in\Z^{\ast}$ ne postoji najveæi zajednièki višekratnik.

\noindent\newline{\bf Dokaz.} Pretpostavimo da je $m$ zajednièki višekratnik od $a$ i $b$ i to najveæi takav. No, uzmemo li $m'=m |a|$, vrijedi da $m'|m$ i $m'|a$, pa tako i $m'|b$. Stoga je $m'$ zajednièki višekratnik od $a$ i $b$. No, kako je $m'=m|a|$, vrijedi $m\leq m'$ što je u suprotnosti s pretpostavkom da je $m$ najveæi zajednièki višekratnik.

\begin{flushright}
$\square$\\
\end{flushright}

\noindent{\bf Propozicija.} Za svaki $a,b\in\Z^{\ast}$ vrijedi:

\begin{equation*}
\textnormal{lcm}(a,b)=\frac{|a b|}{\gcd(a,b)}.
\end{equation*}

\noindent\newline{\bf Dokaz.} Neka je $l=\textnormal{lcm}(a,b)$ i $l'=\frac{|a b|}{\gcd(a,b)}$. Po definiciji vrijedi $a|l$, $b|l$ i $l\leq l'$ (jer je to najmanji zajednièki višekratnik - ukoliko je $l'$ zajednièki višekratnik). Pokažimo kako je $l'$ uistinu zajednièki višekratnik od $a$ i $b$, tj. da $a|l$ i $b|l$. Uzmimo $g=\gcd(a,b)$. Tada po definiciji najveæeg zajednièkog djelitelja $g|a$ i $g|b$, tj. $\exists x,y\in\Z$ takvi da $a=g x$ i $b=g y$. Uzmemo li $l'=\frac{|a b|}{g}=\frac{|a g x|}{g}$. Kako je $g$ pozitivan, vrijedi $l'=|a x|$ pa $a|l'$. Na isti naèin se pokaže da $b|l'$ pa je $l'$ zajednièki višekratnik od $a$ i $b$. Pokažimo da $l|l'$. Uzmemo li $\frac{l}{l'}$ imamo:

\begin{equation*}
\frac{l}{l'}=\frac{l g}{|a b|}.
\end{equation*}

\noindent\newline Po prethodno dokazanoj propoziciji vrijedi $g=az+bw$ za neki $z,w\in\Z^{\ast}$. Tada je $lg=alz+blw$. Kako $a|l$ i $b|l$ postoje $m,n\in\Z$ takvi da je $l=a m$ i $l=b n$. Stoga je $lg=a b n z+ a b m w=a b(n z+m w)$ pa $(a b)|(l g)$. Stoga je $\frac{l}{l'}=u$, gdje je $u\in\Z$ takav da je $l g=a b u$. Dakle, $l=l' u$, i.e. $l'|l$ pa mora biti $l'\leq l$ (ne moramo paziti na apsolutnu vrijednost jer uzimamo u obzir samo pozitivne višekratnike). Kako je $l\leq l'$ ($l$ je najmanji zajednièki višekratnik) i $l'\leq l$ vrijedi $l=l'$ i time je dokazana propozicija.

\begin{flushright}
$\square$\\
\end{flushright}

\noindent{\bf Korolar.} Ako za $a,b\in\Z^{\ast}$ vrijedi $\gcd(a,b)=1$ tada je $\textnormal{lcm}(a,b)=|a b|$.

\noindent\newline{\bf Dokaz.} Slijedi direktno iz prethodne propozicije. Ako je $\gcd(a,b)=1$, tada je:

\begin{equation*}

$\square$\\
\end{flushright}

\end{flushright}

\noindent{\bf Teorem (Fundamentalni teorem aritmetike).} Svaki se prirodan broj veæi od $2$ može prikazati kao umnožak prostih brojeva (ili je prost broj) na jedinstven naèin do na poredak.

\noindent\newline{\bf Dokaz.} {\it Egzistencija} Dokaz se provodi jakom matematièkom indukcijom. Neka je $n=2$. Tada je $n$ prost broj pa vrijedi baza indukcije. Pretpostavimo da se svaki $k<n$ može prikazati kao umnožak prostih brojeva ili je sam prost broj. Ako je $n$ prost, tada smo gotovi. Ako $n$ nije prost, tada postoji neki $n_1$ i $n_2$, razlièiti od $1$, takvi da je $n=n_1 n_2$. No, $n_1<n$ i $n_2<n$, pa se po pretpostavci indukcije mogu prikazati kao umnošci prostih brojeva.

{\it Jedinstvenost.} Pretpostavimo da je $p_1 p_2\cdots p_n=q_1 q_2\cdots q_m$, gdje su $m,n\in\N$ i $m>n$. Po Euklidovoj lemi, ukoliko podijelimo cijeli izraz s $p_1$, i kako su svi $p_i$ i $q_j$ prosti, mora biti da je $p_1$ jednak nekom $q_j$ (bez smanjenja opæenitosti možemo pretpostaviti kako je to upravo $q_1$). Tako dijeleæi sa svakim $p_i$ (isto, bez smanjenja opæenitosti možemo pretpostaviti da æe odgovarati upravo broju $q_i$), dobivamo $1=q_{n+1}q_{n+2}\cdots q_m$. No, tada $q_{n+1},q_{n+2},\ldots, q_m$ moraju biti jednaki $1$. Po tome imamo $p_i=q_i$, za svaki $i\in\{1,\ldots,n\}$ dok su ostali $q_j=1$ i time nebitni.

\begin{flushright}
$\square$\\
\end{flushright}

\noindent{\bf Lema.} Neka su $p\neq q$ prosti brojevi te $i,j\in\N$. Tada je $\gcd{(p^i,q^j)}=1$.

\noindent\newline{\bf Dokaz.} Oèito je kako $p^i=\underbrace{p\cdot p\cdots p}_{i\textnormal{ puta}}$ i $q^j=\underbrace{q\cdot q\cdots q}_{j\textnormal{ puta}}$ ne sadrže nijedan zajednièki djelitelj.

\begin{flushright}
$\square$\\
\end{flushright}

\noindent\newline{\bf Dokaz.} Kako $p^i|n$, postoji $k\in\N$ (radimo u skupu prirodnih brojeva, i $p^i$ i $n$ su prirodni pa mora biti i $k$) takav da je $n=p^i k$. Takoðer, za $q^j|n$ postoji $l\in\N$ takav da je $n=q^j l$. Primijetimo kako odavdje trivijalno slijedi $p^i\nmid k$ i $q^j\nmid l$. Takoðer, bitna opaska jeste da su $p^i$ i $q^j$ relativno prosti. Imamo $p^i k=q^j l$. Ukoliko podijelimo jednadžbu s $p^i$, po Euklidovoj lemi (i prethodnoj opasci) mora biti $p^i|l$. Za neki $l'\in\N$ je $l=l'p^i$. Tada je $n=q^j l' p^i$, za neki $l'\in\N$, pa stoga $p^i q^j|n$.

\begin{flushright}
$\square$\\
\end{flushright}

\noindent{\bf Teorem (Euklidov teorem).} Postoji beskonaèno mnogo prostih brojeva.

\noindent\newline{\bf Dokaz.} Pretpostavimo da postoji $n$ prostih brojeva $p_1,p_2,\ldots,p_n$. Tada postoji složen broj $a=p_1 p_2\cdots p_n$. Uzmimo $b=a+1$. \begin{flushright}
$\square$\\

\noindent{\bf Definicija.} Neka su $a,b\in\Z$, $m\in\N\backslash\{1\}$. Kažemo kako je $a$ kongruentan $b$ modulo $m$ i pišemo $a\equiv b\pmod m$ ako vrijedi $m|(a-b)$.

\noindent\newline{\bf Primjer.} Oèito je $17\equiv 2\pmod 5$ jer $5|(17-2)$. Isto tako i $2\equiv 17\pmod 5$ jer $5|(2-17)$. Slièno, $-17\equiv -2\pmod 5$ i $-2\equiv -17\pmod 5$, ali nije $-17\equiv 2\pmod 5$ i nije $-2\equiv 17\pmod 5$. Ipak bi bilo npr. $-13\equiv 2\pmod 5$.

\noindent\newline{\bf Primjedba.} Iako jeste $15\equiv 0\pmod 5$ u biti ekvivalentno izjavi $5|15$, ipak nije preporuèljivo gledati na to u istom svjetlu.

\noindent\newline{\bf Propozicija.} Relacija kongruencije je relacija ekvivalencije.

\noindent\newline{\bf Dokaz.} (i) {\it Refleksivnost.} Znamo kako $m|0$ jer za $0\in\Z$ vrijedi $0=0\cdot m$. Kako je, oèito, $a-a=0$ za svaki $a\in\Z$, vrijedi da $m|(a-a)$. Tako $a\equiv a\pmod m$ što dokazuje reflektivnost relacije kongruencije.

(ii) {\it Simetriènost.} Ukoliko imamo $a\equiv b\pmod m$ to znaèi da $m|(a-b)$, dakle postoji $k\in\Z$ takav da je $a-b=m k$. To je $-(b-a)=m k$ pa pomnoživši sve s $-1$ dobivamo $b-a=m\cdot(-k)$. Stoga postoji $k'\in\Z$, $k'=-k$ takav da $b-a=k'm$ pa $m|(b-a)$. Tj. $b\equiv a\pmod m$. Simetriènost, dakle, vrijedi.

\noindent\newline{\bf Dokaz.} Za svaki $k\in\Z$ definiramo $x_k=a+k m$. Tada, uvrstimo to u kongruenciju i dobivamo $x_k\equiv a\pmod m$, tj. $a+k m\equiv a\pmod m$. Po prethodnoj propoziciji to je ekvivalentno $a\equiv a\pmod m$.

\noindent{\bf Propozicija.} Neka je $m\in\N$ i $a,b\in\Z$ takvi da je $\gcd{(a,m)}=1$. Linearna kongruencija $a x\equiv b\pmod m$ tada ima jedinstveno rješenje modulo $m$.

\noindent\newline{\bf Dokaz.} {\it Postojanje.} Kako je $\gcd{(a,m)}=1$, po Bezoutovoj lemi postoje $a',m'\in\Z$ takvi da je $a'a+m'm=1$, tj. $m'm=1-a'a$. To je ekvivalentno, po definiciji kongruencije, $1\equiv a'a\pmod m$. Uzmemo li dakle $a x\equiv b\pmod m$ i pomnožimo li tu kongruenciju s dobivenim $a'\in\Z$, dobivamo $a'a x\equiv a'b\pmod m$. No, $a'a\equiv 1\pmod m$ pa imamo $x\equiv a'b\pmod m$.

{\it Jedinstvenost.} Pretpostavimo kako postoji $x'\in\Z$ takav da je $a x'\equiv b\pmod m$, ali $x'$ nekongruentno s $x_0$. To znaèi kako $m\nmid x'-x_0$ tj. po teoremu o dijeljenju s ostatkom postoje $q,r\in\Z$, takvi da je $x'-x_0=q m+r$, gdje je $0\leq r<|m|=m$. Stoga je $x'-x_0\equiv r\pmod m$. No, obzirom da je $a x'\equiv b\pmod m$, dobivamo $x'\equiv a_0 b\pmod m$ (po prethodnom postupku). No $x_0=a_0 b$ pa imamo $0\equiv r\pmod m$, tj. $r=m r'$, za neki $r'\in\Z$. Stoga imamo $x'-x_0=q m+m r'$, tj. $x'-x_0=m(q+r')$. Dakle, $x'\equiv x_0\pmod m$, što je u kontradikciji s pretpostavkom da $m\nmid x'-x_0$.

\begin{flushright}
$\square$\\
\end{flushright}

\noindent{\bf Teorem (Kineski teorem o ostatcima).} Neka su $a,b\in\Z$ i $m,n\in\Z^{+}$ takvi da je $\gcd{(m,n)}=1$. Tada sustav $x\equiv a\pmod m$, $x\equiv b\pmod n$ ima jedinstveno rješenje modulo $m n$.

\noindent\newline{\bf Dokaz.} {\it Postojanje.} Kako je $\gcd{(m,n)}=1$, po Bezoutovoj lemi postoje $p,q\in\Z$ takvi da je $m p+n q=1$. Odatle slijedi $m p\equiv 1\pmod n$ i $n q\equiv 1\pmod m$. Uzmemo li $x_0=m p b+n q a$, lako se vidi kako æe takav $x_0$ zadovoljavati obje kongruencije. Prvo, $m p b+n q a\equiv a\pmod m$ daje $m p b+n q a-a=m s$, za neki $s\in\Z$. Iz toga slijedi $n q a-a=m(s-p b)$, tj. $n q a\equiv a\pmod m$. No, zbog $n q\equiv 1\pmod m$, imamo $a\equiv a\pmod m$. Slièno, $m p b+n q a\equiv b\pmod n$ daje $m p b+n q a-b=n t$, za neki $t\in\Z$. Tada je $m p b-b=n(t-q a)$ što je ekvivalentno kongruenciji $m p b\equiv b\pmod n$. Zbog $m p\equiv 1\pmod n$ imamo $b\equiv b\pmod n$. Dakle, $x_0=m p b+n q a$ je rješenje sustava, a brojevi $p$ i $q$ se mogu dobiti pomoæu Euklidovog algoritma.

{\it Jedinstvenost.} Pretpostavimo kako postoji i $y_0\in\Z$ takav da je $y_0\equiv a\pmod m$ i $y_0\equiv b\pmod n$. Iz toga slijedi $x_0\equiv y_0\pmod m$ i $x_0\equiv y_0\pmod n$. To znaèi kako $m|x_0-y_0$ i $n|x_0-y_0$. Iz prvog uvjeta dobivamo da postoji $q\in\Z$ takav da je $x_0-y_0=m q$. No, kako $n|x_0-y_0$, tj. $n|m q$, obzirom da je $\gcd{(m,n)}=1$, po Euklidovoj lemi imamo $n|q$, tj. postoji $p\in\Z$ takav da je $m q=m n p$. Stoga je $x_0-y_0=m n p$, tj. $x_0\equiv y_0\pmod{m n}$.

\begin{flushright}
$\square$\\
\end{flushright}

\noindent{\bf Korolar.} Neka je $k\in\Z^{+}-\{1\}$ i neka su $a_1,\ldots,a_k\in\Z$ i $m_1,\ldots,m_k\in\Z$ takvi da je $\gcd{(m_i,m_j)}=1$, za svaki $i\neq j$, $i,j\in\{1,\ldots,k\}$. Tada sustav $x\equiv a_1\pmod{m_1},\ldots,x\equiv a_k\pmod{m_k}$ ima jedinstveno rješenje modulo $m_1\cdots m_k$.

\noindent\newline{\bf Dokaz.} Provodi se po indukciji. Neka je $k=2$. Tada sustav $x\equiv a_1\pmod{m_1}$, $x\equiv a_2\pmod{m_2}$ po prethodnom teoremu ima jedinstveno rješenje $x_0$ modulo $m_1 m_2$. Prepostavimo kako tvrdnja vrijedi za neki $k\in\Z$. \begin{equation*}

\end{equation*}

\noindent\newline Vidimo kako æe biti $M q a_{k+1}+m_{k+1}p x_0\equiv a_i\pmod{m_i}$, za svaki $i\in\{1,\ldots,k\}$ jer $M q a_{k+1}+m_{k+1}p x_0-a_i=m_i s_i$, gdje je $s_i\in\Z$. Time je jedinstvenost rješenje modulo $m_1\cdots m_{k+1}$ dokazana i tvrdnja vrijedi za svaki $k\in\Z$.

\noindent{\bf Definicija.} Neka je $x\in\R$. Definiramo $\left\lfloor x\right\rfloor=\max{\{k\in\Z:\ k\leq x\}}$.

\begin{equation*}

\begin{equation*}

\end{equation*}

\noindent{\bf Definicija.} Neka je zadan skup:

\begin{equation*}
\varphi_m=\{n\in\Z^{+}:\ (\gcd{(m,n)}=1)\wedge(n\leq m)\}.
\end{equation*}

\noindent\newline Tada je Eulerova funkcija $\varphi:\Z^{+}\rightarrow\Z^{+}$ definirana kao $\varphi(m)=\left|\varphi_m\right|$.

\noindent\newline{\bf Teorem (Euler\footnote{Ovaj teorem sam dokazao pomoæu teorije grupa u svojoj skripti {\it Abstract Algebra}. Dokaz je, uz poznavanje teorije grupa, vrlo izravan i oèigledan.}).} Neka su $a\in\Z-\{0\}$ i $m\in\Z^{+}-\{1\}$ takvi da je $\gcd{(a,m)}=1$. Tada vrijedi:

\begin{equation*}

\end{equation*}

\noindent\newline{\bf Dokaz.} Neka je $S=\{k\in\Z^{+}\cap[1,m-1]:\ \gcd{(a,k)}=1\}$. Oèito je kako je $|S|=\varphi(m)$ jer za svaka dva $k_1\neq k_2$, $k_1,k_2\in S$, vrijedi $k_1\not\equiv k_2\pmod m$ (ako bi bili jednaki modulo $m$, obzirom da su oba manji od $m$, bilo bi $k_1=k_2$, što je u suprotnosti s pretpostavkom da su razlièiti). Stoga, pomnožimo li $a k_1\equiv a k_2\pmod m$ dobivamo $a x k_1\equiv a x k_2\pmod m$, tj. $k_1\equiv k_2\pmod m$. No, obzirom da su $k_1,k_2\in S$, vrijedi $0<k_1,k_2<m$, pa mora biti $k_1=k_2$, što je u suprotnosti s pretpostavkom da su razlièiti. Stoga vrijedi da su svi elementi u $a S$ razlièiti modulo $m$ pa je $|a S|=|S|=\varphi(m)$. To znaèi kako se svi elementi iz $S$ nalaze u $a S$ modulo $m$ i obratno. \begin{equation*}

\end{equation*}

\noindent\newline Ukoliko preuredimo te elemente s desne strane, dobivamo produkt od $\varphi(m)$ elemenata $a$, pa je:

\begin{equation*}

\end{equation*}

\noindent\newline No, obzirom da je $\gcd{(a_i,m)}=1$, tada je takoðer $\gcd{(A,m)}=1$, gdje je $A=a_1\cdots a_{\varphi(m)}$. Po Bezoutovoj lemi postoje $p,q\in\Z$ takvi da je $A p+m q=1$, što implicira $A p\equiv 1\pmod m$. Stoga, pomožimo li obje strane gornje kongruencije s $p$, dobivamo:

\begin{equation*}

\end{equation*}

\noindent\newline No, to je zbog simetriènosti relacije kongruencije isto što i $a^{\varphi(m)}\equiv 1\pmod m$, što je i trebalo dokazati.

\begin{flushright}
$\square$\\
\end{flushright}

\noindent{\bf Lema.} Neka je $p\in P$ te $m,n\in\Z^{+}$. Tada, $\gcd{(p^m,n)}=1$ ako i samo ako $\gcd{(p,n)}=1$.

\noindent\newline{\bf Dokaz.} {\it Nužnost.} Neka je $\gcd{(p^m,n)}=1$. Tada postoje $x,y\in\Z$ takvi da je $p^m x+n y=1$. Iz toga dobivamo $p(p^{m-1} x)+n y=1$. Pretpostavimo li da $\gcd{(p,n)}=g$ i $g\neq 1$, tada je $p=p' g$ i $n=n' g$. No iz toga slijedi, obzirom da je $p$ prost i $g\neq 1$, kako je $g=p$ pa je $n=n' p$. Stoga imamo $p(p^{m-1}x)+n'p y=1$. Tada je $p\left(p^{m-1}x+n'y\right)=1$ i mora biti $p=1$, što je nemoguæe jer $1\notin P$. Dakle, $\gcd{(p,n)}=1$. {\it Dovoljnost.} Neka je $\gcd{(p,n)}=1$. Pretpostavimo li kako je $\gcd{(p^m,n)}=g$, gdje je $g\neq 1$, tada $g|p^m$ i $g|n$. Iz $g|p^m$, po Euklidovoj lemi, slijedi kako je $g=p^k$, gdje je $1<k\leq m$. \end{flushright}

\noindent{\bf Teorem.} Vrijedi $\varphi(p^m)=p^{m-1}(p-1)$, za svaki $p\in P$ i $m\in\Z^{+}$.

\noindent{\bf Definicija.} Neka je $m\in\Z^{+}-\{1\}$ i $g\in\Z$, $\gcd{(g,m)}=1$. Ako za svaki $a\in\Z$, gdje je $\gcd{(a,m)}=1$, postoji $k\in\Z$ takav da vrijedi $g^k\equiv a\pmod m$, kažemo da je $g$ {\bf primitivan korijen} modulo $m$.

\noindent\newline{\bf Primjedba.} Kroz teoriju grupa, mogli bismo reæi kako je primitivan korijen generator grupe $\zmodinv{m}$, tj. $g\in\zmod{m}$ takav da vrijedi $\cyc{g}=\zmodinv{m}$ (gdje $\cyc{g}$ oznaèava ciklièku grupu generiranu elementom $g$, a ne principalan ideal polja $\zmod{m}$ generiran elementom $g$).

\noindent\newline{\bf Definicija.} Neka je $m\in\Z^{+}-\{1\}$. {\bf Redom elementa} $a\in\Z$ nazivamo nenegativan broj $n=\min{\{k\in\Z^{+}:\ a^k\equiv 1\pmod m\}}$ i pišemo $\ordm{a}{m}=n$. Ukoliko takav broj ne postoji pišemo $\ordm{a}{m}=0$ i kažemo kako je element $a$ {\bf beskonaènog reda}.

\noindent\newline{\bf Propozicija.} Neka je $m\in\Z^{+}-\{1\}$. Tada, $\ordm{a}{m}>0$ ako i samo ako $\gcd{(a,m)}=1$.

\noindent\newline{\bf Dokaz.} {\it Nužnost.} Neka je $\ordm{a}{m}>0$. Tada je $\ordm{a}{m}\neq 0$ i postoji $k\in\Z^{+}$ takav da je $a^k\equiv 1\pmod m$. Po definiciji relacije kongruencije, postoji $q\in\Z$ takav da je $a^k-1=q m$. Pretpostavimo kako je $\gcd{(a,m)}=g>1$. Tada $a=b g$ i $m=n g$, za neke $b,n\in\Z$. Iz $a^k-1=q m$ dobivamo $(b g)^k-1=q(n g)$. To je pak ekvivalentno izrazu $b^k g^k-1=q n g$, tj. $g(g^{k-1} b^k-q n)=1$. Obzirom da je $g\in\Z^{+}$ i $g^{k-1} b^k-q n\in\Z$, može samo biti $g=1$, što je u kontradikciji s pretpostavkom da je $\gcd{(a,m)}>1$ i mora biti $\gcd{(a,m)}=1$. {\it Dovoljnost.} Pretpostavimo kako je $\gcd{(a,m)}=1$. Po Eulerovom teoremu, $a^{\varphi(m)}\equiv 1\pmod m$. No, postoji li moguænost da postoji neki element reda $d$ modulo $p$ a da nije u $C(a)$? Promotrimo jednadžbu $x^d-1\equiv 0\pmod p$. Oèito je kako $(a^k)^d\equiv 1\pmod p$ (jer je $\ordm{a}{p}=d$, pa je $(a^d)^k\equiv 1\pmod p$). Dakle, svi elementi iz $C(a)$ su rješenja kongruencije $x^d-1\equiv 0\pmod p$. No, obzirom da se i dalje radi o polju $\zmod{p}$, polinom $x^d-\overline{1}$ takvih rješenja može imati najviše $d$. Stoga, sva su rješenja u $C(a)$ i ne mogu biti u nijednom drugom skupu. To implicira kako je broj elemenata reda $d$ modulo $p$ toèno $\varphi(d)$.
